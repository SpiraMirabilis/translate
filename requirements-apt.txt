# APT package equivalents for requirements.txt
# For Debian/Ubuntu systems that restrict pip to virtual environments.
#
# Install with:
#   sudo apt install $(grep -v '^#' requirements-apt.txt | grep -v '^$')
#
# Tested against: Ubuntu 24.04 LTS (Noble)
#
# NOTE: The AI provider SDKs (openai, anthropic, google-generativeai) are NOT
# available as apt packages. They must be installed via pip in a virtual
# environment — see the note at the bottom of this file.

# Core dependencies (available via apt)
python3-openai
python3-dotenv

# User interface
python3-questionary
python3-rich
python3-pyperclip

# EPUB processing
python3-ebooklib
python3-bs4
python3-html2text

# HTML parsing
python3-lxml

# Web auth
python3-itsdangerous

# --------------------------------------------------------------------------
# NOT AVAILABLE VIA APT — install these in a venv:
#
#   python3 -m venv --system-site-packages .venv
#   source .venv/bin/activate
#   pip install anthropic google-generativeai
#
# The --system-site-packages flag lets the venv see the apt-installed
# packages above, so you only need pip for the two missing SDKs.
